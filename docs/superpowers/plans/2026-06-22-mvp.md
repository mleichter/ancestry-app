# Ancestry App MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Running family tree app where users can add persons, create relationships, and visualize the tree in a browser.

**Architecture:** FastAPI async backend + PostgreSQL, React 18 SPA frontend, both in Docker Compose. Browser talks directly to backend (CORS). Alembic migrations run at backend container startup via entrypoint script.

**Tech Stack:** Python 3.12, FastAPI ≥ 0.115, SQLAlchemy 2.x + asyncpg, Alembic, PostgreSQL 16, React 18, TypeScript, Vite 5, Tailwind CSS 3, React Flow 11, TanStack Query v5, React Router v6, React Hook Form + Zod.

## Global Constraints

- All API routes under `/api/v1/`
- UUID primary keys generated Python-side with `uuid4()`
- Dates stored as `String(10)` (YYYY, YYYY-MM, or YYYY-MM-DD) for partial date support
- `VITE_API_URL` env var controls frontend→backend URL (default: `http://localhost:8000`)
- Docker Compose: `docker-compose.yml` (prod) + `docker-compose.override.yml` (dev, auto-merged)
- Repo root: `/project/src/ancestry-app`
- UI language: German
- Frontend port: 3000, Backend port: 8000

```python
import asyncio
from logging.config import fileConfig
from sqlalchemy import pool
from sqlalchemy.engine import Connection
from sqlalchemy.ext.asyncio import async_engine_from_config
from alembic import context

from app.models.base import Base
from app.models.media import Media      # noqa: F401 — must import for metadata
from app.models.person import Person    # noqa: F401
from app.models.relationship import Relationship  # noqa: F401
from app.config import get_settings

config = context.config

if config.config_file_name is not None:
    fileConfig(config.config_file_name)

target_metadata = Base.metadata


def get_url() -> str:
    return get_settings().database_url


def run_migrations_offline() -> None:
    context.configure(
        url=get_url(),
        target_metadata=target_metadata,
        literal_binds=True,
        dialect_opts={"paramstyle": "named"},
    )
    with context.begin_transaction():
        context.run_migrations()


def do_run_migrations(connection: Connection) -> None:
    context.configure(connection=connection, target_metadata=target_metadata)
    with context.begin_transaction():
        context.run_migrations()


async def run_async_migrations() -> None:
    cfg = config.get_section(config.config_ini_section, {})
    cfg["sqlalchemy.url"] = get_url()
    connectable = async_engine_from_config(cfg, prefix="sqlalchemy.", poolclass=pool.NullPool)
    async with connectable.connect() as connection:
        await connection.run_sync(do_run_migrations)
    await connectable.dispose()


def run_migrations_online() -> None:
    asyncio.run(run_async_migrations())


if context.is_offline_mode():
    run_migrations_offline()
else:
    run_migrations_online()
```

from uuid import UUID
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from app.models.person import Person
from app.models.relationship import Relationship
from app.schemas.tree import TreeData, TreePersonNode, TreeRelationshipEdge

def _person_to_node(p: Person) -> TreePersonNode:
    return TreePersonNode(
        id=p.id,
        first_name=p.first_name,
        last_name=p.last_name,
        birth_name=p.birth_name,
        date_of_birth=p.date_of_birth,
        date_of_death=p.date_of_death,
        is_living=p.is_living,
        gender=p.gender.value if p.gender else None,
    )

def _rel_to_edge(r: Relationship) -> TreeRelationshipEdge:
    return TreeRelationshipEdge(
        id=r.id,
        person_a_id=r.person_a_id,
        person_b_id=r.person_b_id,
        type=r.type.value,
    )

async def get_full_tree(db: AsyncSession) -> TreeData:
    persons = (await db.execute(select(Person))).scalars().all()
    rels = (await db.execute(select(Relationship))).scalars().all()
    return TreeData(
        nodes=[_person_to_node(p) for p in persons],
        edges=[_rel_to_edge(r) for r in rels],
    )

async def get_subtree(db: AsyncSession, person_id: UUID, depth: int = 3) -> TreeData:
    visited: set[UUID] = {person_id}
    frontier: set[UUID] = {person_id}
    for _ in range(depth):
        if not frontier:
            break
        rels = (
            await db.execute(
                select(Relationship).where(
                    Relationship.person_a_id.in_(frontier)
                    | Relationship.person_b_id.in_(frontier)
                )
            )
        ).scalars().all()
        new_ids = {
            pid
            for r in rels
            for pid in (r.person_a_id, r.person_b_id)
            if pid not in visited
        }
        visited |= new_ids
        frontier = new_ids

    persons = (
        await db.execute(select(Person).where(Person.id.in_(visited)))
    ).scalars().all()
    rels = (
        await db.execute(
            select(Relationship).where(
                Relationship.person_a_id.in_(visited),
                Relationship.person_b_id.in_(visited),
            )
        )
    ).scalars().all()
    return TreeData(
        nodes=[_person_to_node(p) for p in persons],
        edges=[_rel_to_edge(r) for r in rels],
    )

import { useCallback, useMemo } from 'react'
import ReactFlow, {
  Background,
  Controls,
  MiniMap,
  useNodesState,
  useEdgesState,
  type Node,
  type Edge,
} from 'reactflow'
import dagre from 'dagre'
import 'reactflow/dist/style.css'
import PersonNode from './PersonNode'
import type { TreeData, TreeNode as TNode } from '../../types'

const NODE_W = 160
const NODE_H = 80
const nodeTypes = { person: PersonNode }

function applyDagreLayout(nodes: Node[], edges: Edge[]): Node[] {
  const g = new dagre.graphlib.Graph()
  g.setDefaultEdgeLabel(() => ({}))
  g.setGraph({ rankdir: 'TB', ranksep: 80, nodesep: 40 })
  nodes.forEach(n => g.setNode(n.id, { width: NODE_W, height: NODE_H }))
  edges.forEach(e => g.setEdge(e.source, e.target))
  dagre.layout(g)
  return nodes.map(n => {
    const { x, y } = g.node(n.id)
    return { ...n, position: { x: x - NODE_W / 2, y: y - NODE_H / 2 } }
  })
}

function treeDataToFlow(data: TreeData): { nodes: Node[]; edges: Edge[] } {
  const nodes: Node[] = data.nodes.map(n => ({
    id: n.id,
    type: 'person',
    position: { x: 0, y: 0 },
    data: n as TNode,
  }))

  const edges: Edge[] = data.edges.map(e => ({
    id: e.id,
    source: e.person_a_id,
    target: e.person_b_id,
    animated: e.type === 'partner',
    style: {
      stroke: e.type === 'parent_child' ? '#6366f1' : '#ec4899',
      strokeWidth: 2,
    },
    label: e.type === 'partner' ? '♥' : undefined,
  }))

  const laid = applyDagreLayout(nodes, edges)
  return { nodes: laid, edges }
}

export default function GraphView({ data }: { data: TreeData }) {
  const { nodes: initNodes, edges: initEdges } = useMemo(() => treeDataToFlow(data), [data])
  const [nodes, , onNodesChange] = useNodesState(initNodes)
  const [edges, , onEdgesChange] = useEdgesState(initEdges)

  return (
    <ReactFlow
      nodes={nodes}
      edges={edges}
      onNodesChange={onNodesChange}
      onEdgesChange={onEdgesChange}
      nodeTypes={nodeTypes}
      fitView
      fitViewOptions={{ padding: 0.2 }}
    >
      <Background />
      <Controls />
      <MiniMap nodeColor={n => {
        const g = (n.data as TNode).gender
        return g === 'male' ? '#bfdbfe' : g === 'female' ? '#fbcfe8' : '#e5e7eb'
      }} />
    </ReactFlow>
  )
}
